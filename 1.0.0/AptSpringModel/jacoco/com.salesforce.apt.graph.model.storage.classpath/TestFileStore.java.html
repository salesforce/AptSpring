package com.salesforce.apt.graph.model.storage.classpath;

import java.io.File;
import java.io.FileOutputStream;
import java.io.IOException;
import java.io.OutputStream;

import com.salesforce.apt.graph.model.DefinitionModel;
import com.salesforce.apt.graph.model.storage.DefinitionOutputStreamProvider;

public class TestFileStore implements DefinitionOutputStreamProvider {

  private final File outputDirectory;
    
  public TestFileStore(File rootDirectory) {
    outputDirectory = new File(rootDirectory, FileStore.STANDARD.getPath() + File.separator);
  }
  
  @Override
  public OutputStream store(DefinitionModel model) throws IOException {
    File output = new File(outputDirectory, model.getIdentity() + ".json");
    File parentDir = output.getParentFile();
    if (!(parentDir.exists() 
          && parentDir.isDirectory()
          && parentDir.canWrite())
        && !parentDir.mkdirs()) {
      throw new IOException("could not write directories for model storage");
    }
    return new FileOutputStream(output);
  }
}
