91    @Test
92    public void testSimpleRoundTrip() throws IOException {
93      File root = testFolder.newFolder();
94      DefinitionModelStore definitionStore = getDefinitionModelStore(root);
95      DefinitionModel model = new DefinitionModel(SIMPLE_MODEL);
96      model.addDependencyNameToSha256("test1", "ABCDEF1234567890");
97      definitionStore.store(model);
98      List<DefinitionModel> foundModels = definitionStore.lookup(SIMPLE_MODEL);
99      assertThat(foundModels).hasSize(1);
100     assertThat(foundModels.get(0).getSha256()).isNotNull();
101     assertThat(foundModels.get(0).getIdentity()).isEqualTo(model.getIdentity());
102     assertThat(foundModels.get(0).getSha256()).isEqualTo(model.getSha256());
103     assertThat(foundModels.get(0).getDependencyNameToSha256()).containsEntry("test1", "ABCDEF1234567890");
104     cleanUp(root);
105   }

108   @Test
109   public void testSha256AndSourceLocationAreNotStored() throws IOException {
110     File root = testFolder.newFolder();
111     TestFileStore fileStore = new TestFileStore(root);
112     ClassLoader classLoader = new URLClassLoader(new URL[] {root.toURI().toURL()},
113         Thread.currentThread().getContextClassLoader());
114     ResourceLoader loader = new ClasspathUrlResourceLoader(classLoader);
115     DefinitionModelStore definitionStore = new GsonDefinitionModelStore(loader, fileStore);
116     DefinitionModel model = new DefinitionModel(SIMPLE_MODEL);
117     definitionStore.store(model);
118 
119     List<Resource> resources = loader.getEntries(SIMPLE_MODEL);
120     assertThat(resources).hasSize(1);
121     assertThat(toString(resources.get(0).getInputStream()))
122        .doesNotContain("sha256")
123        .doesNotContain("sourceLocation")
124        .contains("\"elementLocation\": \"simple.Model\"");
125     cleanUp(root);
126   }

180   private String toString(InputStream inputStream) throws IOException {
181     ByteArrayOutputStream result = new ByteArrayOutputStream();
182     byte[] buffer = new byte[1024];
183     int length;
184     while ((length = inputStream.read(buffer)) != -1) {
185       result.write(buffer, 0, length);
186     }
187     result.flush();
188     result.close();
189     return result.toString(StandardCharsets.UTF_8.name());
190   }

194   private void cleanUp(File file) throws IOException {    
195     Files.walkFileTree(file.toPath(), new SimpleFileVisitor<Path>() {
196       @Override
197       public FileVisitResult visitFile(Path file, BasicFileAttributes attrs) throws IOException {
198         Files.delete(file);
199         return FileVisitResult.CONTINUE;
200       }
201       
202       @Override
203       public FileVisitResult postVisitDirectory(Path dir, IOException exc) throws IOException {
204         Files.delete(dir);
205         return FileVisitResult.CONTINUE;
206       }
207     });
