27  package com.salesforce.apt.graph.model.storage.classpath;
28  
29  import java.io.File;
30  import java.io.IOException;
31  import java.net.MalformedURLException;
32  import java.net.URL;
33  import java.net.URLClassLoader;
34  import java.util.ArrayList;
35  import java.util.Enumeration;
36  import java.util.List;
37  
38  import com.salesforce.apt.graph.model.storage.Resource;
39  import com.salesforce.apt.graph.model.storage.ResourceLoader;
40  
41  public class ClasspathUrlResourceLoader implements ResourceLoader {
42  
43    private final ClassLoader classloader;
44    
45    public ClasspathUrlResourceLoader() {
46      classloader = Thread.currentThread().getContextClassLoader();
47    }
48  
49    public ClasspathUrlResourceLoader(ClassLoader classLoader) {
50      classloader = classLoader;
51    }
52  
53    /**
54     * If the file exists, use it a root of the UrlClassLoader
55     * @param root file system directory to include as part of the classloader.
56     */
57    public ClasspathUrlResourceLoader(File root) {
58      try {
59        classloader = new URLClassLoader(new URL[] {root.toURI().toURL()},
60            Thread.currentThread().getContextClassLoader());
61      } catch (MalformedURLException | NullPointerException ex) {
62        throw new IllegalArgumentException("Invalide File, it's not a url.", ex);
63      }
64    }
65    
66    /**
67     * Finds all possible resources that match the name of the DefinitionMode we wish to find.
68     */
69    public List<Resource> getEntries(String name) {
70      List<Resource> output  = new ArrayList<>();
71      try {
72        Enumeration<URL> resources = classloader.getResources(FileStore.STANDARD.getPath() + "/" 
73            + name + ".json");
74        while (resources.hasMoreElements()) {
75          output.add(new UrlResource(resources.nextElement()));
76        }
77      } catch (IOException ioe) {
78        throw new RuntimeException("Could not read classpath while looking for ForceInject models of name: " + name, ioe);
79      }
80      return output;
81    }
82  
83  }
