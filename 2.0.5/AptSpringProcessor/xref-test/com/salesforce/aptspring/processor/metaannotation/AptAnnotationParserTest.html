38  import org.junit.Test;

40  import com.google.testing.compile.JavaFileObjects;
41  import com.salesforce.apt.graph.model.storage.classpath.FileStore;
42  import com.salesforce.aptspring.processor.VerifiedSpringConfiguration;
43  
44  public class AptAnnotationParserTest {
45    
46    @Test
47    public void testAnnoProcMetaAnnotations() throws IOException {
48      JavaFileObject annotation = JavaFileObjects.forSourceLines(
49              "test.BeanWrapper",
50              "package test;",
51              "",
52              "import org.springframework.beans.factory.annotation.Qualifier;",
53              "import org.springframework.context.annotation.Bean;",
54              "import org.springframework.core.annotation.AliasFor;",
55              "import java.lang.annotation.Documented;",
56              "import java.lang.annotation.ElementType;",
57              "import java.lang.annotation.Retention;",
58              "import java.lang.annotation.RetentionPolicy;",
59              "import java.lang.annotation.Target;",
60              "",
61              "@Target({ElementType.METHOD, ElementType.ANNOTATION_TYPE})",
62              "@Retention(RetentionPolicy.RUNTIME)",
63              "@Documented",
64              "@Bean",
65              "public @interface BeanWrapper {",
66              " ",
67              "  @AliasFor(annotation=Bean.class, value=\"name\")",
68              "  String[] otherName() default \"\";",
69              "",
70              "}"
71      );
72      
73      JavaFileObject definitionClass = JavaFileObjects.forSourceLines(
74              "test.TestClass",
75              "package test;",
76              "",
77              "import org.springframework.beans.factory.annotation.Qualifier;",
78              "import org.springframework.context.annotation.Bean;",
79              "import org.springframework.context.annotation.Configuration;",
80              "",
81              "  @com.salesforce.aptspring.Verified",
82              "  public class TestClass {",
83              "",
84              "    @BeanWrapper(otherName = \"value1\")",
85              "    public String value1(@Qualifier(\"value2\") String x) { return \"\";}",
86              "",
87              "    @BeanWrapper(otherName = \"value2\")",
88              "    public String value2() { return \"\";}",
89              "",
90              "}");
91      
92      assertAbout(javaSources())
93              .that(Arrays.asList(annotation, definitionClass))
94              .processedWith(new VerifiedSpringConfiguration())
95              .compilesWithoutError()
96              .and()
97              .generatesFileNamed(StandardLocation.SOURCE_OUTPUT, "test", "TestClass_" + FileStore.STANDARD.getPath() + ".java")
98              .and()
99              .generatesFileNamed(StandardLocation.CLASS_OUTPUT, "test", "TestClass_" + FileStore.STANDARD.getPath() + ".class");
100   }
101   
102   @Test
103   public void testAnnoProcMetaAnnotation3() throws IOException {
104     JavaFileObject definitionClass = JavaFileObjects.forSourceLines(
105             "test.TestClass",
106             "package test;",
107             "",
108             "import org.springframework.beans.factory.annotation.Qualifier;",
109             "import org.springframework.context.annotation.Bean;",
110             "import org.springframework.context.annotation.Configuration;",
111             "",
112             "  @com.salesforce.aptspring.Verified",
113             "  public class TestClass {",
114             "",
115             "    @Override",
116             "    public String value1(@Qualifier(\"value2\") String x) { return \"\";}",
117             "",
118             "    @Bean",
119             "    public String value2() { return \"\";}",
120             "",
121             "}");
122     
123     assertAbout(javaSources())
124             .that(Arrays.asList(definitionClass))
125             .processedWith(new VerifiedSpringConfiguration())
126             .failsToCompile()
127             .withErrorContaining("All methods on @Configuration must have @Bean annotation")
128             .in(definitionClass)
129             .onLine(11)
130             .and()
131             .withErrorContaining("All @Bean annotations must define at least one name for a bean.")
132             .in(definitionClass)
133             .onLine(14);
134             
135   }
